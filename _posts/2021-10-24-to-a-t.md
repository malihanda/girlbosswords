    <ul>
    <li>GAMEKEEPER --> GATEKEEPER --> M</li>
    <li>MINISERIES --> MINISTRIES --> E</li>
    <li>BLACK MARKER --> BLACK MARKET --> R</li>
    <li>ENCOURAGE --> ENTOURAGE --> C</li>
    <li>WASHED AWAY --> WASTED AWAY --> H</li>
    </ul>
